OADT value typing

This corresponds to the auxiliary oblivious value typing relation in Fig. 10 in the paper. ovalty v œâ means v is an oblivious value of oblivious type value œâ. This is essentially a subset of typing, but we have it so that the dynamic semantics does not depend on typing.
Inductive ovalty : expr -> expr -> Prop :=
| OTUnit : ovalty <{ () }> <{ ùüô }>
| OTOBool b : ovalty <{ [b] }> <{ ~ùîπ }>
| OTProd v1 v2 œâ1 œâ2 :
    ovalty v1 œâ1 -> ovalty v2 œâ2 ->
    ovalty <{ (v1, v2) }> <{ œâ1 * œâ2 }>
| OTOSum b v œâ1 œâ2 :
    ovalty v <{ ite b œâ1 œâ2 }> ->
    (* Make sure the unused oblivious type is a value. *)
    otval <{ ite b œâ2 œâ1 }> ->
    ovalty <{ [inj@b<œâ1 ~+ œâ2> v] }> <{ œâ1 ~+ œâ2 }>

Evaluation context (‚Ñá)

(* This style is inspired by Iron Lambda. Maybe I should try other encoding
style later. This one can be quite annoying for proof automation. *)
This corresponds to evaluation contexts ‚Ñá in Fig. 17 which extends ‚Ñá in Fig. 10 in the paper. We define evaluation context ‚Ñá as the hole-filling function. ‚Ñá e fills the hole in ‚Ñá with e. ectx ‚Ñá asserts that ‚Ñá is a well-formed context. Due to reusing the same syntax for both term-level and type-level application, application is evaluated from right to left for convenience.

| CtxOSum1 œÑ2 : ectx (fun œÑ1 => <{ œÑ1 ~+ œÑ2 }>)
| CtxOSum2 œâ1 : otval œâ1 -> ectx (fun œÑ2 => <{ œâ1 ~+ œÑ2 }>)
| CtxApp1 v2 : wval v2 -> ectx (fun e1 => <{ e1 v2 }>)
| CtxApp2 e1 : ectx (fun e2 => <{ e1 e2 }>)
| CtxLet e2 : ectx (fun e1 => <{ let e1 in e2 }>)
| CtxSec : ectx (fun e => <{ sùîπ e }>)
| CtxIf e1 e2 : ectx (fun e0 => <{ if e0 then e1 else e2 }>)
| CtxOIf1 e1 e2 : ectx (fun e0 => <{ ~if e0 then e1 else e2 }>)
| CtxOIf2 v0 e2 : wval v0 -> ectx (fun e1 => <{ ~if v0 then e1 else e2 }>)

This corresponds to the step relation in Fig. 17 which extends the step relation in Fig. 10 in the paper.
Inductive step : expr -> expr -> Prop :=
| SApp l œÑ e v :
    wval v ->
    <{ (\:{l}œÑ => e) v }> -->! <{ e^v }>
| SLet v e :
    wval v ->
    <{ let v in e }> -->! <{ e^v }>
| SIf b e1 e2 :
    <{ if b then e1 else e2 }> -->! <{ ite b e1 e2 }>
| SCase b œÑ v e1 e2 :
    wval v ->
    <{ case inj@b<œÑ> v of e1 | e2 }> -->! <{ ite b (e1^v) (e2^v) }>
| SProj b v1 v2 :
    wval v1 -> wval v2 ->
    <{ œÄ@b (v1, v2) }> -->! <{ ite b v1 v2 }>
| SUnfold X X' v :
    wval v ->
    <{ unfold<X> (fold <X'> v) }> -->! v
| SFun x T e :
    Œ£ !! x = Some (DFun T e) ->
    <{ gvar x }> -->! <{ e }>
| SOADT X œÑ e v :
    wval v ->
    Œ£ !! X = Some (DOADT œÑ e) ->
    <{ (gvar X) v }> -->! <{ e^v }>
| SSec b :
    <{ sùîπ b }> -->! <{ [b] }>
| SOInj b œâ v :
    otval œâ -> oval v ->
    <{ ~inj@b<œâ> v }> -->! <{ [inj@b<œâ> v] }>
| SMux b v1 v2 :
    wval v1 -> wval v2 ->
    <{ mux [b] v1 v2 }> -->! <{ ite b v1 v2 }>
(* One of the most interesting rules. *)
| SOCase b œâ1 œâ2 v e1 e2 v1 v2 :
    oval v ->
    ovalty v1 œâ1 -> ovalty v2 œâ2 ->
    <{ ~case [inj@b<œâ1 ~+ œâ2> v] of e1 | e2 }> -->!
      <{ ~if [b] then (ite b (e1^v) (e1^v1)) else (ite b (e2^v2) (e2^v)) }>
| STapeOIf b v1 v2 :
    woval v1 -> woval v2 ->
    <{ tape (~if [b] then v1 else v2) }> -->! <{ mux [b] (tape v1) (tape v2) }>
| STapePair v1 v2 :
    woval v1 -> woval v2 ->
    <{ tape (v1, v2) }> -->! <{ (tape v1, tape v2) }>
(* tape v is a no-op if v is an oblivious value (except for pair). Spell
out all the cases here for determinism and proof convenience. *)
| STapeUnitV :
    <{ tape () }> -->! <{ () }>
| STapeBoxedLit b :
    <{ tape [b] }> -->! <{ [b] }>
| STapeBoxedInj b œâ v :
    otval œâ -> oval v ->
    <{ tape [inj@b<œâ> v] }> -->! <{ [inj@b<œâ> v] }>
| SOIf b v1 v2 ‚Ñá :
    lectx ‚Ñá ->
    wval v1 -> wval v2 ->
    ‚Ñá <{ ~if [b] then v1 else v2 }> -->! <{ ~if [b] then ,(‚Ñá v1) else ,(‚Ñá v2) }>
(* Step under evaluation context *)
| SCtx ‚Ñá e e' :
    ectx ‚Ñá ->
    e -->! e' ->
    ‚Ñá e -->! ‚Ñá e'

where "e '-->!' e'" := (step e e').

End step.
