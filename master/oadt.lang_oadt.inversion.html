Ltac tsf_kinding ctor kinding_inv :=
  lazymatch ctor with
  | KSub => tsf_skip
  | _ =>
      let H := fresh in
      pose proof ctor as H;

Lemma kinding_inv_complete Œ£ Œì œÑ Œ∫ :
  Œ£; Œì ‚ä¢ œÑ :: Œ∫ ->
  kinding_inv Œ£ Œì œÑ Œ∫.
Proof.
  induction 1; eauto using kinding_inv, join_ub_l, join_ub_r.

  select (kinding_inv _ _ _ _) (fun H => destruct H);
    econstructor; subst; eauto using top_inv with lattice_naive_solver.
Qed.

Ltac tsf_typing ctor typing_inv :=
  lazymatch ctor with
  (* Remove the cases about TIte and TCase. *)
  | TIteNoDep => tsf_skip
  | TCaseNoDep => tsf_skip
  | TIte => tsf_skip
  | TCase => tsf_skip
  | TConv => tsf_skip
  | _ =>
      let H := fresh in
      pose proof ctor as H;
      repeat
        lazymatch type of H with
        | forall e : ?T, _ =>
            refine (forall e : T, _ : Prop); specialize (H e)
        | ?Œ£; ?Œì ‚ä¢ ?e :{?l} ?œÑ =>
            let l' := fresh "l'" in
            let œÑ' := fresh "œÑ'" in
            refine (forall (l' : llabel) (œÑ' : expr), _ : Prop);
            lazymatch l with
            | ‚ä§ => refine (l' = ‚ä§ -> _ : Prop)
            | ‚ä• => idtac
            | _ => refine (l ‚äë l' -> _ : Prop)
            end;
            exact (Œ£ ‚ä¢ œÑ' ‚â° œÑ -> typing_inv Œ£ Œì e l' œÑ')
        end
  end.

Lemma typing_inv_complete Œ£ Œì e l œÑ :
  gctx_wf Œ£ ->
  Œ£; Œì ‚ä¢ e :{l} œÑ ->
  typing_inv Œ£ Œì e l œÑ.
Proof.
  intros Hwf.
  induction 1; eauto using typing_inv with equiv_naive_solver.

  (* TIte *)
  - econstructor; eauto;
    lazymatch goal with
    | H : _ ‚ä¢ _ : ?œÑ |- _ =>
        rewrite (open_lc_intro œÑ) by eauto using typing_type_lc
    end; eauto with equiv_naive_solver.

  (* TCase *)
  - econstructor; eauto; intros;
    lazymatch goal with
    | H : _ ‚ä¢ ?œÑ :: _ |- _ =>
        rewrite (open_lc_intro œÑ) by eauto using kinding_lc
    end; eauto with equiv_naive_solver.

  (* TConv *)
  - select (typing_inv _ _ _ _ _) (fun H => sinvert H);
    econstructor; subst;
    eauto using (top_inv (A:=bool)) with equiv_naive_solver lattice_naive_solver.
Qed.
